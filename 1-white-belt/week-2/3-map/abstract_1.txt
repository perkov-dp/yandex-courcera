void PrintMap (const map <int, string>& m)
{
	cout << "Size = " << m.size() << endl;
	for (auto item : m)
	{
		cout << item.first << ": " << item.second << endl;
	}
}

//	перернутый map
map <string, int> BuildReversedMap (const map <int, string>& m)
{
	map <int, string> result;
	for (auto item : m)
	{
		result[item.second] = item.first;
	}
	return result;
}
