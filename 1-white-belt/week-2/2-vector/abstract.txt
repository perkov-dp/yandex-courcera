Считывание эл-тов в-ра при известном размере в-ра
int n;
cin >> n;
vector <string> v(n);
for (string& s : v)
	cin >> s;

Если размер в-ра неизвестен:
vector <string> v;
В цикле while поочередно дописываем строки в в-р
int i = 0;	//	номер шага
while (i < n)
{
	string s;
	cin >> s;
	v.push_back(s);
	++i;
}

Если мы заранее знаем содержимое в-ра
vector <int> days_in_month = {31, 28, 31, 30, 31};	//	в-р дней в к-дом месяце

//	заполнение в-ра значениями по умолчанию
vector <int> is_holiday(28, false);

//	расширение в-ра	без изменения текущих данных. Новые ячейки заполняются нулями
is_holiday.resize(31);

//	если мы хотим переиспользовать в-р и проинициализироватьь его новыми значениями
is_holiday.assign(31, false);

//	если н/очистить в-р
is_holiday.clear();
